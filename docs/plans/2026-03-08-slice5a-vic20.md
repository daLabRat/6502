## Phase C: VIC 6560/6561 Video and Audio Chip

### Task C1: Implement the VIC chip

**Files:**
- Write: `crates/vic20/src/vic.rs`

**Background:** The VIC 6560 (NTSC) generates a composite video signal by racing the beam. It has:
- 22 visible text columns × 23 visible rows (expandable)
- 8×8 character cells (or 8×16 in 8-row mode)
- Colors: foreground from color nibble RAM, background/border from registers
- Multicolor mode: 3.5×8 cells, 4 colors per cell (bit 3 of color nibble enables multicolor)
- Audio: 3 square wave oscillators + 1 noise channel, each with on/off and frequency; shared 4-bit volume register
- Pixel clock: 6.5 MHz (NTSC); CPU at 1.022 MHz = 6.5/6.37 (approx 1 MHz for our purposes)

**VIC register map ($9000–$900F):**
```
$9000: [7]=interlace [6:0]=horizontal origin (color clocks from left)
$9001: [7:0]=vertical origin (lines from top)
$9002: [7]=video matrix base bit 9 [6:0]=columns (7 in exp)
$9003: [7]=raster line bit 0 [6:1]=rows [0]=char size (0=8×8, 1=8×16)
$9004: [7:1]=raster line bits 8:1
$9005: [7:4]=char base bits 13:10 [3:0]=video base bits 13:10
$9006: light pen X (read-only)
$9007: light pen Y (read-only)
$9008-$9009: paddle X/Y (read-only)
$900A: [6:0]=voice 1 frequency (alto)
$900B: [6:0]=voice 2 frequency (tenor)
$900C: [6:0]=voice 3 frequency (soprano)
$900D: [6:0]=noise frequency
$900E: [7:4]=auxiliary color [3:0]=volume
$900F: [7:4]=screen color [3:0]=border color
```

**Step 1: Write `crates/vic20/src/vic.rs`:**
```rust
use emu_common::FrameBuffer;

/// NTSC VIC-20 screen dimensions (visible area).
pub const SCREEN_WIDTH: u32  = 22 * 8; // 176 pixels
pub const SCREEN_HEIGHT: u32 = 23 * 8; // 184 pixels

/// VIC 6560/6561 — video and audio chip.
pub struct Vic {
    pub regs: [u8; 16],

    // Raster state
    raster_line: u16,
    raster_x:    u16,
    pub frame_ready: bool,
    pub framebuffer: FrameBuffer,

    // Audio
    tone_counter: [u16; 3],
    tone_state:   [bool; 3],
    noise_lfsr:   u16,
    noise_state:  bool,
    noise_counter: u16,

    // Sample generation
    sample_rate:  u32,
    sample_accum: f64,
    pub sample_buffer: Vec<f32>,
}

/// VIC-20 NTSC colour palette (16 colors, RGB24).
static VIC_PALETTE: [u32; 16] = [
    0x000000, 0xFFFFFF, 0x880000, 0xAAFFEE,
    0xCC44CC, 0x00CC55, 0x0000AA, 0xEEEE77,
    0xDD8855, 0x664400, 0xFF7777, 0x333333,
    0x777777, 0xAAFF66, 0x0088FF, 0xBBBBBB,
];

impl Vic {
    pub fn new() -> Self {
        Self {
            regs: [0; 16],
            raster_line: 0,
            raster_x: 0,
            frame_ready: false,
            framebuffer: FrameBuffer::new(SCREEN_WIDTH, SCREEN_HEIGHT),
            tone_counter: [0; 3],
            tone_state: [false; 3],
            noise_lfsr: 0x1FF,
            noise_state: false,
            noise_counter: 0,
            sample_rate: 44100,
            sample_accum: 0.0,
            sample_buffer: Vec::with_capacity(1024),
        }
    }

    pub fn set_sample_rate(&mut self, rate: u32) { self.sample_rate = rate; }

    pub fn read(&self, reg: u16) -> u8 {
        match reg & 0x0F {
            // Raster line (bits 8:1 in $9004, bit 0 in $9003 bit 7)
            0x03 => (self.regs[3] & 0x7F) | (((self.raster_line & 1) as u8) << 7),
            0x04 => (self.raster_line >> 1) as u8,
            r    => self.regs[r as usize],
        }
    }

    pub fn write(&mut self, reg: u16, val: u8) {
        self.regs[(reg & 0x0F) as usize] = val;
    }

    /// Step one CPU cycle (~1 MHz). Renders pixel output and advances audio.
    pub fn step(&mut self, memory: &[u8], color_ram: &[u8]) {
        // Video: render 6 VIC pixels per CPU cycle (approximate)
        for _ in 0..6 {
            self.render_pixel(memory, color_ram);
            self.raster_x += 1;
            // NTSC: 65 cycles per line = 390 color clocks; approx 65×6=390 horizontal pixels
            if self.raster_x >= 390 {
                self.raster_x = 0;
                self.raster_line += 1;
                if self.raster_line >= 261 {
                    self.raster_line = 0;
                    self.frame_ready = true;
                }
            }
        }

        // Audio: step tone channels and noise
        self.step_audio();
    }

    fn render_pixel(&mut self, memory: &[u8], color_ram: &[u8]) {
        // VIC visible area: columns 0-21, rows 0-22, each cell 8×8
        // Origin registers $9000/$9001 offset the display within the full raster
        let origin_x = (self.regs[0] & 0x7F) as u16 * 2; // color clocks
        let origin_y = self.regs[1] as u16;
        let cols = ((self.regs[2] & 0x7F) as u16).max(1);
        let rows = ((self.regs[3] & 0x7E) >> 1) as u16;
        let char_height = if self.regs[3] & 0x01 != 0 { 16u16 } else { 8u16 };

        // Map raster_x → visible pixel x
        let pixel_x = self.raster_x.saturating_sub(origin_x);
        let pixel_y = self.raster_line.saturating_sub(origin_y);

        if pixel_x >= cols * 8 || pixel_y >= rows * char_height { return; }
        if pixel_x >= SCREEN_WIDTH as u16 || pixel_y >= SCREEN_HEIGHT as u16 { return; }

        let col = (pixel_x / 8) as usize;
        let row = (pixel_y / char_height) as usize;
        let cell_x = (pixel_x % 8) as usize;
        let cell_y = (pixel_y % char_height) as usize;

        // Video matrix base: regs[5] bits 3:0 = bits 13:10 (shifted left 10)
        let vm_base = ((self.regs[5] & 0x0F) as usize) << 10;
        let screen_addr = vm_base + row * cols as usize + col;
        let char_code = memory.get(screen_addr).copied().unwrap_or(0) as usize;

        // Character ROM base: regs[5] bits 7:4 = bits 13:10
        let char_base = ((self.regs[5] >> 4) as usize) << 10;
        let char_addr = char_base + char_code * char_height as usize + cell_y;
        let char_byte = memory.get(char_addr).copied().unwrap_or(0);

        // Color nibble RAM: low nibble = foreground color, bit 3 = multicolor enable
        let color_nibble = color_ram.get(row * cols as usize + col).copied().unwrap_or(0);
        let fg_color = color_nibble & 0x07;
        let multicolor = color_nibble & 0x08 != 0;

        let bg_color  = (self.regs[0xF] >> 4) & 0x0F; // screen color
        let aux_color = (self.regs[0xE] >> 4) & 0x0F; // auxiliary color

        let pixel_color = if multicolor && cell_x < 7 {
            // Multicolor: 2 bits/pixel, 4 colors. Effective width = 3.5 pixels = treat as 4-pixel wide
            let pair = (char_byte >> (6 - (cell_x & !1))) & 0x03;
            match pair {
                0 => bg_color,
                1 => (self.regs[0xF] & 0x0F) as u8, // border color
                2 => fg_color,
                _ => aux_color,
            }
        } else {
            // Standard: 1 bit/pixel
            if char_byte & (0x80 >> cell_x) != 0 { fg_color } else { bg_color }
        };

        let rgb = VIC_PALETTE[pixel_color as usize & 0x0F];
        self.framebuffer.set_pixel_rgb(pixel_x as u32, pixel_y as u32, rgb);
    }

    fn step_audio(&mut self) {
        // Tone channels: frequency = cpu_freq / (127 - freq_reg) / 2
        // Step at CPU rate, generate audio at sample rate
        for ch in 0..3 {
            let freq_reg = self.regs[0xA + ch] & 0x7F;
            if freq_reg > 0 && self.regs[0xA + ch] & 0x80 != 0 {
                // Bit 7 = channel on/off
                let period = (127u16).saturating_sub(freq_reg as u16).max(1);
                if self.tone_counter[ch] == 0 {
                    self.tone_counter[ch] = period;
                    self.tone_state[ch] = !self.tone_state[ch];
                } else {
                    self.tone_counter[ch] -= 1;
                }
            }
        }

        // Noise: 16-bit LFSR
        let noise_freq = self.regs[0xD] & 0x7F;
        if noise_freq > 0 && self.regs[0xD] & 0x80 != 0 {
            let noise_period = (127u16).saturating_sub(noise_freq as u16).max(1);
            if self.noise_counter == 0 {
                self.noise_counter = noise_period;
                let bit = (self.noise_lfsr ^ (self.noise_lfsr >> 2) ^ (self.noise_lfsr >> 3) ^ (self.noise_lfsr >> 5)) & 1;
                self.noise_lfsr = (self.noise_lfsr >> 1) | (bit << 15);
                self.noise_state = self.noise_lfsr & 1 != 0;
            } else {
                self.noise_counter -= 1;
            }
        }

        // Generate sample
        let cpu_freq: f64 = 1_022_727.0;
        self.sample_accum += self.sample_rate as f64;
        if self.sample_accum >= cpu_freq {
            self.sample_accum -= cpu_freq;
            let vol = (self.regs[0xE] & 0x0F) as f32 / 15.0;
            let mut sample = 0.0f32;
            for ch in 0..3 {
                if self.regs[0xA + ch] & 0x80 != 0 && self.tone_state[ch] {
                    sample += vol / 4.0;
                }
            }
            if self.regs[0xD] & 0x80 != 0 && self.noise_state {
                sample += vol / 4.0;
            }
            self.sample_buffer.push(sample.min(1.0));
        }
    }

    pub fn drain_samples(&mut self, out: &mut [f32]) -> usize {
        let n = out.len().min(self.sample_buffer.len());
        out[..n].copy_from_slice(&self.sample_buffer[..n]);
        self.sample_buffer.drain(..n);
        n
    }
}
```

**Background:** VIC-20 cartridges are ROM images loaded at specific addresses:

```rust
pub struct Cartridge {
    pub data: Vec<u8>,
    pub load_addr: u16,
}

impl Cartridge {
    /// Load a .prg file (2-byte little-endian load address + data).
    pub fn from_prg(data: &[u8]) -> Result<Self, String> {
        if data.len() < 3 { return Err("PRG too short".into()); }
        let addr = u16::from_le_bytes([data[0], data[1]]);
        Ok(Self { data: data[2..].to_vec(), load_addr: addr })
    }

    /// Load a raw ROM image at a fixed address (for .crt or raw binary).
    pub fn from_rom(data: &[u8], addr: u16) -> Self {
        Self { data: data.to_vec(), load_addr: addr }
    }
}
```

## Phase E: Bus and Memory Map

### Task E1: Implement `Vic20Bus`

**Files:**
- Write: `crates/vic20/src/bus.rs`

**VIC-20 memory map (unexpanded):**
```
$0000-$03FF  RAM (zero page, stack, screen RAM default $1E00)
$0400-$0FFF  RAM (remaining 3KB internal)
$1000-$1FFF  Character ROM (4KB, via VIC addressing, not CPU directly)
$2000-$7FFF  Expansion RAM space (empty unexpanded)
$8000-$8FFF  Expansion ROM (cartridge block 0)
$9000-$900F  VIC chip registers
$9110-$911F  VIA1 (user port, IEC bus input)
$9120-$912F  VIA2 (keyboard, serial, cassette)
$9400-$97FF  Color nibble RAM (4-bit wide, 1KB addressable = 512 nibbles)
$A000-$BFFF  BASIC ROM (8KB)
$C000-$DFFF  Expansion ROM / cartridge (8KB)
$E000-$FFFF  KERNAL ROM (8KB)
```

**Step 1: Write `crates/vic20/src/bus.rs`:**
```rust
use emu_common::Bus;
use crate::vic::Vic;
use crate::via::Via;
use crate::iec_bus::IecBus;

pub struct Vic20Bus {
    pub ram:        [u8; 0x1000], // 4KB internal RAM ($0000-$0FFF)
    pub char_rom:   Vec<u8>,      // 4KB character ROM
    pub basic_rom:  Vec<u8>,      // 8KB BASIC ROM
    pub kernal_rom: Vec<u8>,      // 8KB KERNAL ROM
    pub color_ram:  [u8; 512],    // 4-bit color nibble RAM (512 nibbles = $9400-$97FF)
    pub vic:        Vic,
    pub via1:       Via,          // User port / IEC
    pub via2:       Via,          // Keyboard / cassette
    pub iec_bus:    IecBus,
    // Cartridge ROM (if any)
    cart_rom:    Vec<u8>,
    cart_addr:   u16,
}

impl Vic20Bus {
    pub fn new(basic: Vec<u8>, kernal: Vec<u8>, chargen: Vec<u8>) -> Self {
        Self {
            ram:        [0; 0x1000],
            char_rom:   chargen,
            basic_rom:  basic,
            kernal_rom: kernal,
            color_ram:  [0; 512],
            vic:        Vic::new(),
            via1:       Via::new(),
            via2:       Via::new(),
            iec_bus:    IecBus::new(),
            cart_rom:   Vec::new(),
            cart_addr:  0xA000,
        }
    }

    pub fn load_cart(&mut self, data: Vec<u8>, addr: u16) {
        self.cart_rom = data;
        self.cart_addr = addr;
    }

    pub fn load_prg(&mut self, data: &[u8], addr: u16) {
        for (i, &b) in data.iter().enumerate() {
            self.write_ram(addr.wrapping_add(i as u16), b);
        }
    }

    fn write_ram(&mut self, addr: u16, val: u8) {
        if (addr as usize) < self.ram.len() {
            self.ram[addr as usize] = val;
        }
    }

    fn read_cart(&self, addr: u16) -> u8 {
        if self.cart_rom.is_empty() { return 0xFF; }
        let offset = addr.wrapping_sub(self.cart_addr) as usize;
        self.cart_rom.get(offset).copied().unwrap_or(0xFF)
    }

    /// Provide 16KB view of memory for VIC chip (character + screen RAM access).
    /// VIC sees a 16KB bank based on $9005 bits — simplified: always bank starting at $0000.
    pub fn vic_memory(&self) -> VicMemoryView<'_> {
        VicMemoryView { bus: self }
    }
}

/// A read-only memory view for the VIC chip's internal address bus.
pub struct VicMemoryView<'a> {
    bus: &'a Vic20Bus,
}

impl VicMemoryView<'_> {
    pub fn read(&self, addr: usize) -> u8 {
        // VIC's 16KB bank 0: $0000-$3FFF
        // Character ROM appears at $1000-$1FFF in this bank
        match addr {
            0x0000..=0x0FFF => self.bus.ram[addr],
            0x1000..=0x1FFF => self.bus.char_rom.get(addr - 0x1000).copied().unwrap_or(0),
            _ => 0,
        }
    }
}

impl Bus for Vic20Bus {
    fn read(&mut self, addr: u16) -> u8 {
        match addr {
            0x0000..=0x0FFF => self.ram[addr as usize],
            0x9000..=0x900F => self.vic.read(addr - 0x9000),
            0x9110..=0x911F => self.via1.read(addr - 0x9110),
            0x9120..=0x912F => self.via2.read(addr - 0x9120),
            0x9400..=0x97FF => {
                let idx = (addr - 0x9400) as usize / 2;
                self.color_ram.get(idx).copied().unwrap_or(0) & 0x0F
            }
            0xA000..=0xBFFF => self.basic_rom.get((addr - 0xA000) as usize).copied().unwrap_or(0xFF),
            0xC000..=0xDFFF if !self.cart_rom.is_empty() && self.cart_addr == 0xC000 => self.read_cart(addr),
            0xA000..=0xBFFF if !self.cart_rom.is_empty() && self.cart_addr == 0xA000 => self.read_cart(addr),
            0xE000..=0xFFFF => self.kernal_rom.get((addr - 0xE000) as usize).copied().unwrap_or(0xFF),
            _ => 0xFF,
        }
    }

    fn write(&mut self, addr: u16, val: u8) {
        match addr {
            0x0000..=0x0FFF => self.ram[addr as usize] = val,
            0x9000..=0x900F => self.vic.write(addr - 0x9000, val),
            0x9110..=0x911F => self.via1.write(addr - 0x9110, val),
            0x9120..=0x912F => self.via2.write(addr - 0x9120, val),
            0x9400..=0x97FF => {
                let idx = (addr - 0x9400) as usize / 2;
                if let Some(c) = self.color_ram.get_mut(idx) { *c = val & 0x0F; }
            }
            _ => {}
        }
    }

    fn peek(&self, addr: u16) -> u8 {
        match addr {
            0x0000..=0x0FFF => self.ram[addr as usize],
            0xA000..=0xBFFF => self.basic_rom.get((addr - 0xA000) as usize).copied().unwrap_or(0xFF),
            0xE000..=0xFFFF => self.kernal_rom.get((addr - 0xE000) as usize).copied().unwrap_or(0xFF),
            _ => 0xFF,
        }
    }

    fn tick(&mut self, cycles: u8) {
        // Collect VIC memory before calling step (avoids borrow conflict)
        // Build a flat 16KB snapshot for the VIC chip each tick
        let mut vic_mem = [0u8; 0x4000];
        for i in 0..0x1000 { vic_mem[i] = self.ram[i]; }
        for i in 0..self.char_rom.len().min(0x1000) { vic_mem[0x1000 + i] = self.char_rom[i]; }

        let color_ram_copy = self.color_ram;
        for _ in 0..cycles {
            self.vic.step(&vic_mem, &color_ram_copy);
            self.via1.step();
            self.via2.step();
        }
    }

    fn poll_nmi(&mut self) -> bool { false }
    fn poll_irq(&mut self) -> bool {
        self.via1.irq_pending() || self.via2.irq_pending()
    }
}
```

**Step 1: Write `crates/vic20/src/lib.rs`:**
```rust
pub mod bus;
pub mod cartridge;
pub mod iec_bus;
pub mod vic;
pub mod via;

use emu_common::{AudioSample, Button, CpuDebugState, DebugSection, FrameBuffer, InputEvent, SystemEmulator};
use emu_cpu::Cpu6502;
use bus::Vic20Bus;

pub struct Vic20 {
    pub cpu: Cpu6502<Vic20Bus>,
}

impl Vic20 {
    pub fn with_roms(basic: Vec<u8>, kernal: Vec<u8>, chargen: Vec<u8>) -> Self {
        let bus = Vic20Bus::new(basic, kernal, chargen);
        let mut cpu = Cpu6502::new(bus);
        cpu.bcd_enabled = true;
        cpu.reset();
        Self { cpu }
    }

    pub fn load_prg(&mut self, data: &[u8]) -> Result<(), String> {
        if data.len() < 3 { return Err("PRG too short".into()); }
        let addr = u16::from_le_bytes([data[0], data[1]]);
        self.cpu.bus.load_prg(&data[2..], addr);
        // Set BASIC pointers so BASIC knows where program ends
        // Program end = addr + len - 2 (skip load address header)
        let end = addr.wrapping_add(data.len() as u16 - 2);
        self.cpu.bus.write(0x2D, end as u8);       // VARTAB low
        self.cpu.bus.write(0x2E, (end >> 8) as u8); // VARTAB high
        Ok(())
    }

    pub fn load_cart(&mut self, data: Vec<u8>, addr: u16) {
        self.cpu.bus.load_cart(data, addr);
        self.cpu.reset();
    }
}

impl SystemEmulator for Vic20 {
    fn step_frame(&mut self) -> usize {
        self.cpu.bus.vic.frame_ready = false;
        while !self.cpu.bus.vic.frame_ready {
            self.cpu.step();
        }
        0
    }

    fn framebuffer(&self) -> &FrameBuffer { &self.cpu.bus.vic.framebuffer }

    fn audio_samples(&mut self, out: &mut [AudioSample]) -> usize {
        self.cpu.bus.vic.drain_samples(out)
    }

    fn handle_input(&mut self, event: InputEvent) {
        // VIC-20 keyboard is a 8×8 matrix through VIA2
        // Map common buttons to joystick port (VIA1 Port B)
        let mask = match event.button {
            Button::Up    => Some(0x04u8),
            Button::Down  => Some(0x08),
            Button::Left  => Some(0x10),
            Button::Right => Some(0x80),
            Button::Fire | Button::A => Some(0x20),
            _ => None,
        };
        if let Some(m) = mask {
            if event.pressed {
                self.cpu.bus.via1.irb &= !m;
            } else {
                self.cpu.bus.via1.irb |= m;
            }
        }
    }

    fn reset(&mut self) { self.cpu.reset(); }

    fn set_sample_rate(&mut self, rate: u32) {
        self.cpu.bus.vic.set_sample_rate(rate);
    }

    fn display_width(&self)  -> u32 { vic::SCREEN_WIDTH }
    fn display_height(&self) -> u32 { vic::SCREEN_HEIGHT }
    fn target_fps(&self)     -> f64 { 60.0 }
    fn system_name(&self)    -> &str { "VIC-20" }
    fn save_state_system_id(&self) -> &str { "VIC20" }

    fn cpu_state(&self) -> CpuDebugState {
        CpuDebugState { pc: self.cpu.pc, sp: self.cpu.sp, a: self.cpu.a,
            x: self.cpu.x, y: self.cpu.y, flags: self.cpu.p.bits(), cycles: self.cpu.total_cycles }
    }
    fn peek_memory(&self, addr: u16) -> u8 { self.cpu.bus.peek(addr) }
    fn disassemble(&self, addr: u16) -> (String, u16) {
        emu_cpu::disassemble_6502(|a| self.cpu.bus.peek(a), addr)
    }
    fn step_instruction(&mut self) { self.cpu.step(); }

    fn system_debug_panels(&self) -> Vec<DebugSection> {
        let vic = &self.cpu.bus.vic;
        let regs = &vic.regs;
        let via2 = &self.cpu.bus.via2;
        let vic_sec = DebugSection::new("VIC")
            .row("Raster",   format!("{}", vic.raster_line))
            .row("Screen",   format!("{}cols x {}rows  char={}px",
                                     regs[2] & 0x7F, (regs[3] & 0x7E) >> 1,
                                     if regs[3] & 1 != 0 { 16 } else { 8 }))
            .row("Colors",   format!("bg=${:X} border=${:X} aux=${:X}",
                                     regs[0xF] >> 4, regs[0xF] & 0xF, regs[0xE] >> 4))
            .row("Volume",   format!("{}", regs[0xE] & 0xF))
            .row("Voice1",   format!("freq=${:02X} {}", regs[0xA] & 0x7F, if regs[0xA] & 0x80 != 0 { "on" } else { "off" }))
            .row("Voice2",   format!("freq=${:02X} {}", regs[0xB] & 0x7F, if regs[0xB] & 0x80 != 0 { "on" } else { "off" }))
            .row("Voice3",   format!("freq=${:02X} {}", regs[0xC] & 0x7F, if regs[0xC] & 0x80 != 0 { "on" } else { "off" }))
            .row("Noise",    format!("freq=${:02X} {}", regs[0xD] & 0x7F, if regs[0xD] & 0x80 != 0 { "on" } else { "off" }));
        let via_sec = DebugSection::new("VIA2 (Kbd)")
            .row("ORA/DDRA", format!("${:02X}/${:02X}", via2.ora, via2.ddra))
            .row("ORB/DDRB", format!("${:02X}/${:02X}", via2.orb, via2.ddrb));
        vec![vic_sec, via_sec]
    }
}
```
